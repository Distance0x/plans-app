# 数据库设计

## 数据库选型
- **数据库**: SQLite3
- **ORM**: Drizzle ORM
- **Node.js 绑定**: better-sqlite3

## 数据库 Schema

### 1. tasks 表（任务表）

```sql
CREATE TABLE tasks (
  id TEXT PRIMARY KEY,                    -- UUID
  title TEXT NOT NULL,                    -- 任务标题
  description TEXT,                       -- 任务描述
  priority TEXT CHECK(priority IN ('high', 'medium', 'low')) DEFAULT 'medium',  -- 优先级
  status TEXT CHECK(status IN ('todo', 'in_progress', 'completed')) DEFAULT 'todo',  -- 状态
  due_date TEXT,                          -- 截止日期 (ISO 8601)
  due_time TEXT,                          -- 截止时间 (HH:MM)
  created_at TEXT NOT NULL,               -- 创建时间 (ISO 8601)
  updated_at TEXT NOT NULL,               -- 更新时间 (ISO 8601)
  completed_at TEXT,                      -- 完成时间 (ISO 8601)
  parent_id TEXT,                         -- 父任务 ID（用于子任务）
  order_index INTEGER DEFAULT 0,          -- 排序索引
  estimated_pomodoros INTEGER DEFAULT 0,  -- 预计番茄钟数
  actual_pomodoros INTEGER DEFAULT 0,     -- 实际番茄钟数
  FOREIGN KEY (parent_id) REFERENCES tasks(id) ON DELETE CASCADE
);

-- 索引
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_due_date ON tasks(due_date);
CREATE INDEX idx_tasks_parent_id ON tasks(parent_id);
CREATE INDEX idx_tasks_created_at ON tasks(created_at);
```

### 2. tags 表（标签表）

```sql
CREATE TABLE tags (
  id TEXT PRIMARY KEY,                    -- UUID
  name TEXT UNIQUE NOT NULL,              -- 标签名称
  color TEXT DEFAULT '#3B82F6',          -- 标签颜色 (HEX)
  created_at TEXT NOT NULL                -- 创建时间
);

-- 索引
CREATE INDEX idx_tags_name ON tags(name);
```

### 3. task_tags 表（任务标签关联表）

```sql
CREATE TABLE task_tags (
  task_id TEXT NOT NULL,                  -- 任务 ID
  tag_id TEXT NOT NULL,                   -- 标签 ID
  PRIMARY KEY (task_id, tag_id),
  FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE,
  FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
);

-- 索引
CREATE INDEX idx_task_tags_task_id ON task_tags(task_id);
CREATE INDEX idx_task_tags_tag_id ON task_tags(tag_id);
```

### 4. pomodoro_sessions 表（番茄钟记录表）

```sql
CREATE TABLE pomodoro_sessions (
  id TEXT PRIMARY KEY,                    -- UUID
  task_id TEXT,                           -- 关联任务 ID（可为空）
  session_type TEXT CHECK(session_type IN ('work', 'short_break', 'long_break')) NOT NULL,  -- 会话类型
  duration INTEGER NOT NULL,              -- 时长（秒）
  start_time TEXT NOT NULL,               -- 开始时间 (ISO 8601)
  end_time TEXT,                          -- 结束时间 (ISO 8601)
  completed BOOLEAN DEFAULT 0,            -- 是否完成
  interrupted BOOLEAN DEFAULT 0,          -- 是否中断
  created_at TEXT NOT NULL,               -- 创建时间
  FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE SET NULL
);

-- 索引
CREATE INDEX idx_pomodoro_task_id ON pomodoro_sessions(task_id);
CREATE INDEX idx_pomodoro_start_time ON pomodoro_sessions(start_time);
CREATE INDEX idx_pomodoro_completed ON pomodoro_sessions(completed);
```

### 5. settings 表（设置表）

```sql
CREATE TABLE settings (
  key TEXT PRIMARY KEY,                   -- 设置键
  value TEXT NOT NULL,                    -- 设置值 (JSON)
  updated_at TEXT NOT NULL                -- 更新时间
);
```

### 6. avatar_states 表（虚拟形象状态表）

```sql
CREATE TABLE avatar_states (
  id TEXT PRIMARY KEY,                    -- UUID
  state_type TEXT NOT NULL,               -- 状态类型 (working, resting, celebrating, idle)
  expression TEXT NOT NULL,               -- 表情名称
  message TEXT,                           -- 提示消息
  triggered_at TEXT NOT NULL,             -- 触发时间
  duration INTEGER DEFAULT 0              -- 持续时长（秒）
);

-- 索引
CREATE INDEX idx_avatar_triggered_at ON avatar_states(triggered_at);
```

## Drizzle ORM Schema 定义

```typescript
// electron/database/schema.ts
import { sqliteTable, text, integer, index } from 'drizzle-orm/sqlite-core';

export const tasks = sqliteTable('tasks', {
  id: text('id').primaryKey(),
  title: text('title').notNull(),
  description: text('description'),
  priority: text('priority', { enum: ['high', 'medium', 'low'] }).default('medium'),
  status: text('status', { enum: ['todo', 'in_progress', 'completed'] }).default('todo'),
  dueDate: text('due_date'),
  dueTime: text('due_time'),
  createdAt: text('created_at').notNull(),
  updatedAt: text('updated_at').notNull(),
  completedAt: text('completed_at'),
  parentId: text('parent_id').references(() => tasks.id, { onDelete: 'cascade' }),
  orderIndex: integer('order_index').default(0),
  estimatedPomodoros: integer('estimated_pomodoros').default(0),
  actualPomodoros: integer('actual_pomodoros').default(0),
}, (table) => ({
  statusIdx: index('idx_tasks_status').on(table.status),
  dueDateIdx: index('idx_tasks_due_date').on(table.dueDate),
  parentIdIdx: index('idx_tasks_parent_id').on(table.parentId),
  createdAtIdx: index('idx_tasks_created_at').on(table.createdAt),
}));

export const tags = sqliteTable('tags', {
  id: text('id').primaryKey(),
  name: text('name').notNull().unique(),
  color: text('color').default('#3B82F6'),
  createdAt: text('created_at').notNull(),
}, (table) => ({
  nameIdx: index('idx_tags_name').on(table.name),
}));

export const taskTags = sqliteTable('task_tags', {
  taskId: text('task_id').notNull().references(() => tasks.id, { onDelete: 'cascade' }),
  tagId: text('tag_id').notNull().references(() => tags.id, { onDelete: 'cascade' }),
}, (table) => ({
  pk: { columns: [table.taskId, table.tagId] },
  taskIdIdx: index('idx_task_tags_task_id').on(table.taskId),
  tagIdIdx: index('idx_task_tags_tag_id').on(table.tagId),
}));

export const pomodoroSessions = sqliteTable('pomodoro_sessions', {
  id: text('id').primaryKey(),
  taskId: text('task_id').references(() => tasks.id, { onDelete: 'set null' }),
  sessionType: text('session_type', { enum: ['work', 'short_break', 'long_break'] }).notNull(),
  duration: integer('duration').notNull(),
  startTime: text('start_time').notNull(),
  endTime: text('end_time'),
  completed: integer('completed', { mode: 'boolean' }).default(false),
  interrupted: integer('interrupted', { mode: 'boolean' }).default(false),
  createdAt: text('created_at').notNull(),
}, (table) => ({
  taskIdIdx: index('idx_pomodoro_task_id').on(table.taskId),
  startTimeIdx: index('idx_pomodoro_start_time').on(table.startTime),
  completedIdx: index('idx_pomodoro_completed').on(table.completed),
}));

export const settings = sqliteTable('settings', {
  key: text('key').primaryKey(),
  value: text('value').notNull(),
  updatedAt: text('updated_at').notNull(),
});

export const avatarStates = sqliteTable('avatar_states', {
  id: text('id').primaryKey(),
  stateType: text('state_type').notNull(),
  expression: text('expression').notNull(),
  message: text('message'),
  triggeredAt: text('triggered_at').notNull(),
  duration: integer('duration').default(0),
}, (table) => ({
  triggeredAtIdx: index('idx_avatar_triggered_at').on(table.triggeredAt),
}));
```

## 数据查询示例

### 获取今日任务
```typescript
const today = new Date().toISOString().split('T')[0];
const todayTasks = await db
  .select()
  .from(tasks)
  .where(eq(tasks.dueDate, today))
  .orderBy(desc(tasks.priority), asc(tasks.orderIndex));
```

### 获取未完成任务
```typescript
const incompleteTasks = await db
  .select()
  .from(tasks)
  .where(ne(tasks.status, 'completed'))
  .orderBy(desc(tasks.priority), asc(tasks.dueDate));
```

### 获取任务的标签
```typescript
const taskWithTags = await db
  .select({
    task: tasks,
    tag: tags,
  })
  .from(tasks)
  .leftJoin(taskTags, eq(tasks.id, taskTags.taskId))
  .leftJoin(tags, eq(taskTags.tagId, tags.id))
  .where(eq(tasks.id, taskId));
```

### 获取今日番茄钟统计
```typescript
const today = new Date().toISOString().split('T')[0];
const stats = await db
  .select({
    count: count(),
    totalDuration: sum(pomodoroSessions.duration),
  })
  .from(pomodoroSessions)
  .where(
    and(
      eq(pomodoroSessions.completed, true),
      eq(pomodoroSessions.sessionType, 'work'),
      like(pomodoroSessions.startTime, `${today}%`)
    )
  );
```

## 数据备份策略

### 自动备份
- 每日自动备份数据库文件
- 保留最近 7 天的备份
- 备份文件命名: `plans-app-backup-YYYY-MM-DD.db`

### 手动备份
- 用户可手动触发备份
- 导出为 SQLite 文件
- 支持导入备份文件

### 备份存储位置
```
Windows: %APPDATA%/plans-app/backups/
```
